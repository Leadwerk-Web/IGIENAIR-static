    const mapEl = document.getElementById('admin1_map');

    function getLocationKey(stateId) {
      if (!stateId) return null;
      return STATE_TO_LOCATION[stateId] || stateId;
    }

    // Tek bir lokasyon kartý HTML'i
    function renderLocationCard(data) {
      let html = '<div class="state-card">';
      html += '<h3>' + data.title + '</h3>';

      // Adres + ikon
      html += '<div class="state-detail-row">';
      html +=   '<div class="state-detail-icon icon-location"></div>';
      html +=   '<div class="state-detail-text">';
      data.address.forEach(function (line) {
        html += '<p>' + line + '</p> ';
      });
      html +=   '</div>';
      html += '</div>';

      // Telefon + ikon
      html += '<div class="state-detail-row">';
      html +=   '<div class="state-detail-icon icon-phone"></div>';
      html +=   '<div class="state-detail-text"><p>' + data.phone + '</p> </div>';
      html += '</div>';

      // Mail + ikon
      html += '<div class="state-detail-row">';
      html +=   '<div class="state-detail-icon icon-mail"></div>';
      html +=   '<div class="state-detail-text">';
      html +=     '<p><a href="mailto:' + data.email + '">' + data.email + '</a></p> ';
      html +=   '</div>';
      html += '</div>';

      html += '</div>'; // .state-card

      return html;
    }

    // Detaylarý çiz
    function renderDetails(stateId) {
      // Alle Standorte (boţ value) › 8 lokasyonun hepsini listele
      if (!stateId) {
        let html = '';
        STATES_WITH_INFO.forEach(function (sid) {
          const locKey = getLocationKey(sid);
          const data = locKey && LOCATIONS[locKey] ? LOCATIONS[locKey] : null;
          if (data) {
            html += renderLocationCard(data);
          }
        });

        // Her ihtimale karţý hiçbir lokasyon yoksa fallback metni
        if (!html) {
          html = '<h3>' + defaultText.title + '</h3>';
          defaultText.lines.forEach(function (line) {
            html += '<p>' + line + '</p> ';
          });
        }

        stateDetailsEl.innerHTML = html;
        return;
      }

      // Tek eyalet seçimi
      const locKey = getLocationKey(stateId);
      const data = locKey && LOCATIONS[locKey] ? LOCATIONS[locKey] : null;

      if (!data) {
        let html = '<h3>' + defaultText.title + '</h3>';
        defaultText.lines.forEach(function (line) {
          html += '<p>' + line + '</p> ';
        });
        stateDetailsEl.innerHTML = html;
        return;
      }

      stateDetailsEl.innerHTML = renderLocationCard(data);
    }

    // Harita path'i: sm_state_DEBW gibi class'tan bul
    function setActiveState(stateId) {
      if (!mapEl) {
        renderDetails(stateId);
        return;
      }

      // Eski aktifleri temizle
      mapEl.querySelectorAll('.sm_state.is-active').forEach(function (el) {
        el.classList.remove('is-active');
      });

      // Alle Standorte – haritada hiçbir eyalet seçili olmasýn
      if (!stateId) {
        renderDetails('');
        return;
      }

      const selector = '.sm_state.sm_state_' + CSS.escape(stateId);
      const target = mapEl.querySelector(selector);
      if (target) {
        target.classList.add('is-active');
      }

      renderDetails(stateId);
    }

    // class listesinden DE** kodunu çýkar
    function extractStateIdFromElement(el) {
      if (!el) return null;
      const classAttr = el.getAttribute('class') || '';
      const classes = classAttr.split(/\s+/);
      for (let i = 0; i < classes.length; i++) {
        const cls = classes[i];
        if (cls.indexOf('sm_state_') === 0 || cls.indexOf('sm_label_') === 0) {
          const parts = cls.split('_'); // sm_state_DEBW
          return parts[2] || null;
        }
      }
      return null;
    }

    // Bilgisi olan eyaletlere has-info class'ý ekle
    function markStatesWithInfo() {
      if (!mapEl) return;
      STATES_WITH_INFO.forEach(function (stateId) {
        const selector = '.sm_state.sm_state_' + CSS.escape(stateId);
        mapEl.querySelectorAll(selector).forEach(function (el) {
          el.classList.add('has-info');
        });
      });
    }

    markStatesWithInfo();

    // Harita týklandýđýnda
    if (mapEl) {
      mapEl.addEventListener('click', function (e) {
        let stateId = extractStateIdFromElement(e.target);
        if (!stateId && e.target.parentNode) {
          stateId = extractStateIdFromElement(e.target.parentNode);
        }
        if (!stateId) return;

        // Bilgi olmayan eyaletler: hiçbir ţey yapma
        if (STATES_WITH_INFO.indexOf(stateId) === -1) {
          return;
        }

        e.preventDefault();

        if (selectEl) {
          selectEl.value = stateId;
        }

        setActiveState(stateId);
      });
    }

    // Select deđiţince
    if (selectEl) {
      selectEl.addEventListener('change', function () {
        const val = this.value || '';
        setActiveState(val);
      });
    }

    // Ýlk yükleme – varsayýlan Alle Standorte
    const initialState = ""; // Alle Standorte
    if (selectEl) {
      selectEl.value = initialState;
    }
    setActiveState(initialState);
  });
